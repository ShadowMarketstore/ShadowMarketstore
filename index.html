// Configuration de ton client Supabase const SUPABASE_URL = "https://votre-projet.supabase.co"; const SUPABASE_ANON_KEY = "votre-cle-anon-ici"; // Fonction principale de vérification (déclenchée lors du clic sur "Validé") async function verifierCle() { const inputCode = document.getElementById('shdw-input-code').value.trim(); const btnValider = document.getElementById('shdw-btn-valider'); const successBlock = document.getElementById('shdw-success-block'); const btnDownload = document.getElementById('shdw-btn-download'); // Zone de texte pour afficher les erreurs proprement (à ajouter dans ton HTML si tu veux, ou reste en alerte propre) if (!inputCode) { alert("Veuillez entrer un code d'accès."); return; } // Protection : Verrouiller le bouton de validation pendant la requête btnValider.disabled = true; const texteOriginalValider = btnValider.innerText; btnValider.innerText = "⏳ Vérification..."; try { // Requête vers Supabase const response = await fetch(`${SUPABASE_URL}/rest/v1/codes?code=eq.${encodeURIComponent(inputCode)}&select=*,produits(lien_telechargement)`, { method: 'GET', headers: { 'apikey': SUPABASE_ANON_KEY, 'Authorization': `Bearer ${SUPABASE_ANON_KEY}` } }); if (!response.ok) throw new Error("Erreur réseau"); const data = await response.json(); // 1. Si le code n'existe pas dans la base if (!data || data.length === 0) { alert("❌ Code invalide. Veuillez vérifier votre saisie."); btnValider.disabled = false; btnValider.innerText = texteOriginalValider; return; } const donnesCode = data[0]; const urlTelechargement = donnesCode.produits?.lien_telechargement; if (!urlTelechargement) { alert("⚠️ Ce produit n'a pas de lien de téléchargement configuré."); btnValider.disabled = false; btnValider.innerText = texteOriginalValider; return; } // 2. Si le code est valide : Changement d'état visuel immédiat btnValider.innerHTML = "✓ Validé"; btnValider.style.backgroundColor = "#2ea44f"; // Affichage du bloc vert de succès successBlock.style.display = "block"; // Configuration du bouton blanc de secours btnDownload.onclick = function() { declencherTelechargementSecurise(urlTelechargement, btnDownload); }; // 3. 🚀 LE SAINT-GRAAL DU TÉLÉCHARGEMENT AUTOMATIQUE (Anti-Blocage Pop-up) // Au lieu d'un setTimeout qui se fait bloquer par Chrome/Safari, on change la localisation de la fenêtre // ou on crée un iframe invisible synchrone, ce qui est autorisé par les navigateurs mobiles après un clic. const iframe = document.createElement('iframe'); iframe.style.display = 'none'; iframe.src = urlTelechargement; document.body.appendChild(iframe); // Nettoyage de l'iframe juste après setTimeout(() => document.body.removeChild(iframe), 2000); } catch (error) { console.error("Erreur critique :", error); alert("Une erreur est survenue lors de la connexion. Veuillez réessayer."); btnValider.disabled = false; btnValider.innerText = "Validé"; } } // Fonction de téléchargement manuel (si le client clique sur le bouton blanc) function declencherTelechargementSecurise(url, boutonElement) { if (boutonElement.classList.contains('downloading')) return; boutonElement.classList.add('downloading'); boutonElement.disabled = true; const texteOriginal = boutonElement.innerHTML; boutonElement.innerHTML = "⏳ Téléchargement en cours..."; // Utilisation de la méthode iframe pour le clic manuel aussi (plus stable sur les vieux Android) const iframe = document.createElement('iframe'); iframe.style.display = 'none'; iframe.src = url; document.body.appendChild(iframe); setTimeout(() => { document.body.removeChild(iframe); setTimeout(() => { boutonElement.innerHTML = texteOriginal; boutonElement.disabled = false; boutonElement.classList.remove('downloading'); }, 6000); }, 1000); }
